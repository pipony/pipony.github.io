public static void readFile() {  
    String filePath = "IOStreamLearning/file.txt";  
    try {  
        // 1.ä½¿ç”¨FileReaderæ„å»ºBufferedReaderå¯¹è±¡  
        BufferedReader br1 = new BufferedReader(new FileReader(filePath));  
        String line;

        }  
        // 2.ä½¿ç”¨InputStreamReaderæ„å»ºBufferedReaderå¯¹è±¡  
        // FileReaderæ— æ³•æŒ‡å®šè¯»å–ç¼–ç ï¼Œä½†æ˜¯å¯ä»¥å€ŸåŠ©FileInputStream+InputStreamReaderæ¥æŒ‡å®šç¼–ç

        System.out.println("Error reading file: " + e.getMessage());
